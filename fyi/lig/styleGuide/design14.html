    var selected_num = 1;
    function ViewTab(num)
    {
     var tab, con;

     tab = document.getElementById("tab" + selected_num);
     tab.src = "/Images/Btn/btn_Left_menu0" + selected_num + "off.gif";
     tab.state = 0;

     tab = document.getElementById("tab" + num);
     tab.src = "/Images/Btn/btn_Left_menu0" + num + "on.gif";
     tab.state = 1;

     con = document.getElementById("con" + selected_num);
     con.style.display = "none";

     con = document.getElementById("con" + num);
     con.style.display = "";

     selected_num = num;
    }
